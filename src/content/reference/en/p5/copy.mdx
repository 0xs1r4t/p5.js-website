description: >
  <p>Copies pixels from a source image to a region of the canvas.</p>

  <p>The first parameter, <code>srcImage</code>, is the

  <a href="/reference/p5/p5.Image">p5.Image</a> object to blend. The source
  image can be

  the canvas itself or a

  <a href="/reference/p5/p5.Image">p5.Image</a> object. <code>copy()</code> will
  scale pixels from

  the source region if it isn't the same size as the destination region.</p>

  <p>The next four parameters, <code>sx</code>, <code>sy</code>,
  <code>sw</code>, and <code>sh</code> determine the region

  to copy from the source image. <code>(sx, sy)</code> is the top-left corner of
  the

  region. <code>sw</code> and <code>sh</code> are the region's width and
  height.</p>

  <p>The next four parameters, <code>dx</code>, <code>dy</code>,
  <code>dw</code>, and <code>dh</code> determine the region

  of the canvas to copy into. <code>(dx, dy)</code> is the top-left corner of
  the

  region. <code>dw</code> and <code>dh</code> are the region's width and
  height.</p>

    let img;

    async function setup() {
      // Load the image.
      img = await loadImage('/assets/rockies.jpg');

      createCanvas(100, 100);

      // Use the mountains as the background.
      background(img);

      // Copy a region of pixels to another spot.
      copy(img, 7, 22, 10, 10, 35, 25, 50, 50);

      // Outline the copied region.
      stroke(255);
      noFill();
      square(7, 22, 10);

      describe('An image of a mountain landscape. A square region is outlined in white. A larger square contains a pixelated view of the outlined region.');
    }
